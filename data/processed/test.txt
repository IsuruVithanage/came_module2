𑀧𑀭𑀼𑀫𑀓𑀯𑁂𑀮𑀼𑀰𑀼𑀫𑀦𑀰𑀧𑀼𑀢𑀧𑀭𑀼𑀫𑀓𑀯𑁂𑀮𑀼𑀰𑀧𑀼𑀢𑀧𑀭𑀼𑀫𑀓𑀧𑀼𑀰𑀤𑁂𑀯𑀰𑀮𑁂𑀡𑁂𑀤𑁂𑀯𑀦𑀧𑀺𑀬𑀰𑀭𑀛𑀢𑀺𑀰𑀰𑀅𑀬𑀓𑀰𑀮𑁂𑀡𑁂𑀮𑀟𑀓𑁂𑀮𑁂𑀡𑁂𑀰𑀕𑀰
𑀉𑀧𑀰𑀺𑀓𑀢𑀺𑀰𑀬𑀧𑀤𑀕𑀟𑀺
𑀯𑀰𑀢𑁂𑀭𑀳𑀰𑀤𑀺𑀯𑀺𑀳𑀭𑀺𑀬𑀤𑀫𑀕𑀼𑀢-𑀢𑁂𑀭𑀳𑀮𑁂𑀡𑁂
𑀧𑀭𑀼𑀫𑀓𑀧𑀼𑀰𑀳𑀮𑁂𑀡𑁂𑀅𑀕𑀢𑀅𑀦𑀕𑀢𑀘𑀢𑀼𑀤𑀺𑀰𑀰𑀕𑀰
𑀰𑁄𑀩𑀺𑀓𑀦𑀅𑀕𑀓𑀫𑀺𑀦𑀫𑀭𑀼𑀫𑀓𑀦𑀦𑀝𑀘𑀼𑀮𑀳𑀮𑁂𑀡𑁂𑀅𑀕𑀢𑀅𑀦𑀕𑀢𑀘𑀢𑀼𑀤𑀺𑀰𑀰𑀕𑀰𑀦𑀝𑀘𑀼𑀮𑀳𑀛𑀺𑀢𑀦𑀝𑀤𑀫𑀛𑀬𑀰𑀫𑀼𑀤𑀬
𑀲𑀺𑀥𑀫𑁆𑀤𑀼𑀩𑀮𑀕𑀫𑀓𑀳𑀺𑀉𑀧𑀲𑀓𑀉𑀢𑀭𑀳𑀮𑁂𑀡𑁂𑀲𑀕𑀦𑀺𑀬𑀢𑁂
𑀧𑀭𑀼𑀫𑀓𑀉𑀢𑀭𑀧𑀼𑀢𑀧𑀭𑀼𑀫𑀓𑀢𑀺𑀰𑀳𑀮𑁂𑀡𑁂𑀰𑀕𑀰
𑀢𑀺𑀰𑀭𑀓𑀧𑀼𑀢𑀉𑀧𑀰𑀓𑀫𑀳𑀢𑀺𑀰𑀳𑀮𑁂𑀡𑁂
𑀭𑀛𑀦𑀕𑀛𑀺𑀢𑀭𑀛𑀉𑀢𑀺𑀛𑀬𑀅𑀩𑀺𑀅𑀦𑀼𑀭𑀤𑀺𑀘𑀭𑀛𑀉𑀢𑀺𑀘𑀓𑀭𑀧𑀺𑀢𑀰𑁂𑀇𑀫𑀮𑁂𑀡𑀘𑀢𑀼𑀤𑀺𑀰𑀰𑀰𑀕𑀬𑀅𑀕𑀢𑀅𑀕𑀢𑀦𑀧𑀰𑀼𑀯𑀺𑀳𑀭𑀬𑁂𑀅𑀧𑀭𑀺𑀫𑀺𑀢𑀮𑁄𑀓𑀤𑀢𑀼𑀬𑀰𑀢𑀦𑀰𑀺𑀢𑀰𑀼𑀓𑀬𑁂
𑀕𑀫𑀺𑀓𑀧𑀘𑀮𑀧𑀼𑀢𑀕𑀫𑀺𑀓𑀤𑀼𑀝𑀓𑀢𑀺𑀰𑀧𑀼𑀢𑀕𑀫𑀺𑀓𑀅𑀩𑀬𑀳𑀮𑁂𑀡𑁂𑀅𑀕𑀢𑀅𑀦𑀕𑀢𑀘𑀢𑀼𑀤𑀺𑀰𑀰𑀕𑀰
𑀕𑀫𑀺𑀓𑀛𑀺𑀢𑀉𑀧𑀰𑀺𑀓𑀩𑀤𑀬𑀮𑁂𑀡𑁂𑀰𑀕𑀰
𑀕𑀳𑀧𑀢𑀺𑀫𑀳𑀢𑀺𑀰𑀳𑀮𑁂𑀡𑁂𑀰𑀕𑀰

𑀧𑀭𑀼𑀫𑀓𑀅𑀰𑀅𑀤𑁂𑀓𑀯𑁂𑀮𑀼𑀰𑀛𑀬𑀢𑀺𑀰𑀬𑀮𑁂𑀡𑁂

𑀢𑀺𑀰𑀭𑀓𑀺𑀢𑀢𑁂𑀭𑀰𑀅𑀢𑀺𑀯𑀰𑀺𑀓𑀧𑀼𑀰𑀳𑀮𑁂𑀡𑁂

𑀧𑀭𑀼𑀫𑀓𑀰𑀼𑀫𑀦𑀰𑀮𑁂𑀡𑁂𑀰𑀕𑀰
𑀕𑀫𑀺𑀓𑀦𑀕𑀳𑀮𑁂𑀡𑁂𑀰𑀕𑀰𑀤𑀼𑀧𑀝𑀓𑁂
𑀩𑀢𑀉𑀧𑀢𑀺𑀰𑀳𑀦𑀅𑀩𑀦𑀕𑀭𑀺𑀬𑀦𑀰𑀕𑀰
𑀕𑀧𑀢𑀺𑀰𑀺𑀯𑀓𑀼𑀮𑀳𑀮𑁂𑀡𑁂𑀰𑀕𑀰
𑀤𑁄𑀤𑀰𑀛𑀦𑀦𑀘𑀤𑀓𑀮𑁂𑀡𑁂

𑀕𑀧𑀢𑀺𑀯𑁂𑀮𑀼𑀧𑀼𑀢𑀦𑀢𑀺𑀡𑀺𑀩𑀢𑀺𑀓𑀦𑀰𑀕𑀰𑀩𑀢𑀤𑀢𑀳𑀮𑁂𑀡𑁂
𑀩𑀢𑀰𑀼𑀫𑀦𑀳𑀅𑀴𑀼𑀫𑀳𑀰𑀦𑀫𑀰𑀕𑀰
𑀧𑀼𑀕𑀺𑀬𑀦𑀮𑁂𑀡𑁂𑀰𑀕𑀰𑀤𑀺𑀦𑁂
𑀩𑀭𑀢𑀫𑀳𑀢𑀺𑀰
𑀮𑁂𑀡𑁂𑀰𑀕𑀰𑀫𑀳𑀫𑀢𑀢𑀺𑀰𑀳𑀮𑁂𑀡𑁂𑀰𑀕𑀰
𑀕𑀧𑀢𑀺𑀰𑀕𑀭𑀧𑀼𑀢𑀫𑀛𑀺𑀫𑀳

𑀛𑀺𑀢𑀬𑀉𑀧𑀰𑀺𑀓𑀯𑁂𑀅𑀕𑀢𑀅𑀦𑀕𑀢𑀘𑀢𑀼𑀤𑀺𑀰𑀰𑀕𑀰
𑀉𑀧𑀰𑀓𑀰𑀢𑀰𑀮𑁂𑀡𑁂𑀰𑀕𑀺𑀓𑁂
𑀧𑀭𑀼𑀫𑀓𑀰𑀼𑀫𑀦𑀧𑀼𑀢𑀧𑀭𑀼𑀫𑀓𑀳𑀕𑀭𑀳𑀮𑁂𑀡𑁂𑀅𑀕𑀢𑀅𑀦𑀕𑀢𑀘𑀭𑀼𑀤𑀺𑀰𑀰𑀕𑀰𑀤𑀺𑀦𑁂
𑀧𑀭𑁂𑀫𑀓𑀇𑀰𑀺𑀕𑀼𑀢𑀳𑀮𑁂𑀡𑁂𑀰𑀕𑀰
𑀧𑀭𑀼𑀫𑀓𑀢𑀩𑀢𑀺𑀰𑀧𑀼𑀢𑀧𑀭𑀼𑀫𑀓𑀢𑀺𑀰𑀳𑀮𑁂𑀡𑁂𑀰𑀕𑀰
𑀧𑀭𑀼𑀫𑀓𑀲𑀫𑀡𑀓𑀰𑀩𑀕𑀺𑀦𑀺𑀬𑀦𑀧𑀘𑀩𑀢𑀺𑀓𑀦𑀮𑁂𑀡𑁂
